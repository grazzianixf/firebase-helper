# FirebaseHelper

## Example

```
import { FirebaseHelper } from "@grazzianixf/firebase-helper";

const firebaseConfig = {
  apiKey: "*",
  authDomain: "zzzzzzz.firebaseapp.com",
  databaseURL: "https://zzzz.firebaseio.com",
  projectId: "project_123",
  storageBucket: "zzz.appspot.com",
  messagingSenderId: "******",
  appId: "*****",
};

export const firebaseHelper = new FirebaseHelper(firebaseConfig);
```
